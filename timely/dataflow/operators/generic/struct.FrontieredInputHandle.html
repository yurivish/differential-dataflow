#Examples

use timely::dataflow::operators::ToStream;

    (0..10).to_stream(scope)
           .unary_stream(Pipeline, "example", |input, output| {
               input.for_each(|cap, data| {
                   output.session(&cap).give_content(data);
               });
           });
});
